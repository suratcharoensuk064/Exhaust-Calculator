Engine Exhaust Gas Calculator

Lambda (λ)

λ = (CO₂ + CO/2 + O₂ + NOx/2 + (Hcv/4 × 3.5/(3.5+CO/CO₂) − Ocv/2)) / ((1 + Hcv/4 − Ocv/2)(CO₂ + CO + nHC))







Exhaust Gas Density

ρ = PM / RT



Fuel Mass Flow

ṁfuel = ρfuel × V̇fuel


Exhaust Mass Flow

ṁexhaust = ρ × V̇


function calcDensity(){ let P=parseFloat(document.getElementById("P").value) let Tc=parseFloat(document.getElementById("T").value) let T=Tc+273.15 let M=parseFloat(document.getElementById("M").value) let R=8.314 let rho=(P*M)/(R*T) document.getElementById("density_result").innerHTML = "ρ = " + rho.toFixed(3) + " kg/m³" } function calcMfuel(){ let V=parseFloat(document.getElementById("Vfuel").value) let rho=parseFloat(document.getElementById("rhofuel").value) let V_L_min=V/1000 let m=(rho*V_L_min)/60 document.getElementById("mfuel_result").innerHTML = "ṁfuel = " + m.toFixed(6) + " kg/s" } function calcMex(){ let rho=parseFloat(document.getElementById("rho_ex").value) let V=parseFloat(document.getElementById("Vex").value) let m=rho*V document.getElementById("mex_result").innerHTML = "ṁexhaust = " + m.toFixed(3) + " kg/s" }
